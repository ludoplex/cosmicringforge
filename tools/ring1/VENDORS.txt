# Ring 1 Velocity Tools - Portable via Cosmopolitan
#
# All Ring 1 tools compile with cosmocc to APE binaries.
# APE binaries run on Linux, macOS, Windows, FreeBSD, OpenBSD, NetBSD.
#
# See: RING_CLASSIFICATION.md for complete documentation.
# ══════════════════════════════════════════════════════════════════════════════

makeheaders:
  Source: https://www.hwaci.com/sw/mkhdr/
  License: BSD-2-Clause
  Author: D. Richard Hipp (SQLite author)
  Purpose: Auto-generate header files from C source
  Vendored: tools/ring1/makeheaders/makeheaders.c
  Build: make ring1 (uses cosmocc if available)

gengetopt:
  Source: https://ftp.gnu.org/gnu/gengetopt/
  License: GPL-3.0 (generated code has linking exception)
  Purpose: CLI argument parser generator (.ggo → .c)
  Vendored: tools/ring1/gengetopt/ (see README.md)
  Build: make ring1 (uses cosmocc if available)

# ── System Tools (not vendored) ──────────────────────────────────────────────

cppcheck:
  Source: https://cppcheck.sourceforge.io/
  License: GPL-3.0
  Install: apt install cppcheck / brew install cppcheck
  Purpose: Static analysis
  Note: Not yet available as APE, use system install

# ── Compiler Built-ins ───────────────────────────────────────────────────────

AddressSanitizer (ASan):
  Included with: GCC 4.8+, Clang 3.1+, cosmocc
  Purpose: Memory error detection (buffer overflow, use-after-free)
  Build: make sanitize

UndefinedBehaviorSanitizer (UBSan):
  Included with: GCC 4.9+, Clang 3.3+, cosmocc
  Purpose: Undefined behavior detection
  Build: make sanitize

ThreadSanitizer (TSan):
  Included with: GCC 4.8+, Clang 3.2+
  Purpose: Data race detection
  Build: make tsan
  Note: Not compatible with ASan, run separately
